برنامج حساب_المجموع؛

ثابت
    الحد_الاعلى = 100؛

متغير
    رقم_المستخدم، المجموع, العداد: صحيح؛
    رسالة: خيط_رمزي؛

اجراء اطبع_النتيجة(بالقيمة النتيجة: صحيح؛ بالقيمة الرسالة_المخصصة: خيط_رمزي)؛
{
    اطبع(الرسالة_المخصصة، النتيجة)؛
}؛

{
    المجموع = 0؛
    اطبع("الرجاء إدخال رقم صحيح بين 1 و 10:")؛
    اقرا(رقم_المستخدم)؛

    اذا (رقم_المستخدم > 10) فان
    {
        رقم_المستخدم = 10؛
        اطبع("تم تحديد الرقم الى 10 لانه تجاوز الحد المسموح")؛
    }؛

    كرر (العداد = 1 الى رقم_المستخدم)
    {
        المجموع = المجموع + العداد؛
    }؛
    
    رسالة = "المجموع النهائي هو: "؛
    اطبع_النتيجة(المجموع، رسالة)؛
    
    اذا (المجموع > الحد_الاعلى) فان
    {
        اطبع("تنبيه: المجموع تجاوز الحد الاعلى المحدد.")؛
    }؛
}
